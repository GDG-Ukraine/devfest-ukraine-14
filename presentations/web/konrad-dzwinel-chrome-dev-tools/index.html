    "default_title": "Redmine Issues Checker"

  }

Building Chrome Extensions

saving settings

HTML5 localStorage

localStorage.setItem('config', JSON.stringify(config));
//...
var config = JSON.parse( localStorage.getItem('config') );

Chrome synced storage (requires storage permission)

chrome.storage.sync.set({'config': config}, function() {
  message('Settings saved');
});
//...
chrome.storage.sync.get('config', function(config) {
  message('Settings loaded', config);
});

Building Chrome Extensions

managing other extensions (requires managment permission)

Getting information

chrome.management.get("aalnjolghjkk...", function(ext) {
  console.log('extension info', ext.name, ext.icons, ext.enabled);
});

Enabling & disabling

chrome.management.setEnabled("aalnjolghjkk...", true, function() {
  console.log('extension enabled');
});

Listening to events

chrome.management.onInstalled.addListener(function(extension) {
  console.log('extension ' + extension.name + ' installed');
})
